#include <stdio.h>
#include <stdbool.h>

#define MAX 50

int V;

bool isSafe(int v, int graph[MAX][MAX], int color[], int c) {
    for (int i = 0; i < V; i++) {
        if (graph[v][i] && color[i] == c)
            return false;
    }
    return true;
}

bool graphColoringUtil(int graph[MAX][MAX], int m, int color[], int v) {
    if (v == V)
        return true;

    for (int c = 1; c <= m; c++) {
        if (isSafe(v, graph, color, c)) {
            color[v] = c;

            if (graphColoringUtil(graph, m, color, v + 1))
                return true;

            color[v] = 0;
        }
    }

    return false;
}

bool graphColoring(int graph[MAX][MAX], int m) {
    int color[MAX] = {0};

    if (!graphColoringUtil(graph, m, color, 0)) {
        printf("\nNo solution exists.\n");
        return false;
    }

    printf("\nSolution (color of each vertex):\n");
    for (int i = 0; i < V; i++)
        printf("Vertex %d â†’ Color %d\n", i, color[i]);

    return true;
}

int main() {
    int graph[MAX][MAX];
    int m;

    printf("Enter number of vertices: ");
    scanf("%d", &V);

    printf("\nEnter adjacency matrix (%d x %d):\n", V, V);
    for (int i = 0; i < V; i++) {
        for (int j = 0; j < V; j++) {
            scanf("%d", &graph[i][j]);
        }
    }

    printf("\nEnter number of colors: ");
    scanf("%d", &m);

    graphColoring(graph, m);

    return 0;
}
