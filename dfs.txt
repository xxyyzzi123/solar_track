#include <stdio.h>

#define MAX 10

int graph[MAX][MAX], visited[MAX];
int n;  // number of nodes

void resetVisited() {
    for (int i = 0; i < n; i++)
        visited[i] = 0;
}

void DFS(int v) {
    visited[v] = 1;
    printf("%d ", v);

    for (int i = 0; i < n; i++) {
        if (graph[v][i] == 1 && !visited[i])
            DFS(i);
    }
}

int main() {
    int start;

    printf("Enter number of nodes (max %d): ", MAX);
    scanf("%d", &n);

    printf("Enter adjacency matrix (%d x %d):\n", n, n);
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            scanf("%d", &graph[i][j]);

    printf("Enter starting node (0 to %d): ", n - 1);
    scanf("%d", &start);

    resetVisited();
    printf("DFS order: ");
    DFS(start);
    printf("\n");

    return 0;
}
